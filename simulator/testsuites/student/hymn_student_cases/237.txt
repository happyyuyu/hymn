	READ 		#reads in user value for input1
	STOR input1	#stores value as input1
	READ 		#reads in user value for input2
	STOR input2	#stores value as input2
	LOAD input2 #loads input2 value into AC
	JZER final  #if input2 is 0, jumps to load final total
	LOAD input1 #loads input1 value into AC

	LOAD total #loads current total
	ADD input2 #adds input2 to total
	STOR total #stores result in total
	LOAD input1 #loads input1
	SUB one    #subtracts one from input1
	STOR input1 #stores new input1
	JUMP repeat #jumps back to check if input1 is zero
final: 	LOAD total #loads final total
	WRITE #writes total
	HALT #stops program

input2: 0 #user input2 value
total: 0  #multiplication of integers to user input value
